$(".tip").okPopup({
  openEvent   : 'mouseenter',
  closeEvent  : 'mouseleave',
  openEffect  : 'fadeIn',
  closeEffect : 'fadeOut' ,
  where       : {location:'elementTop', offsetTop:-5},
  onOpen      : function(e,popup){ 
    popup.html($(e.currentTarget).data('content')); 
    popup.open(e); 
  }
});

Notifications

Add a regular notification 

            
$("#example-3").on('click',function(e){
  e.preventDefault();
  var notification = $.okPopup.create($(this),{ 
    openEffect : 'fadeIn',
    parent     : "#notification-queue" // Set the parent to a specific element
  })
  .addClass('notification')
  .open(e, $(e.currentTarget).data('content')); 

  setTimeout(function() {
    notification.fadeOut(function(){
      notification.remove();
    });
  }, 3000);

});

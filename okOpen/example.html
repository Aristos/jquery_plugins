Tooltips

Hover Me 
          
$("#example-1").on('mouseenter mouseleave',function(e){
  var tooltip = $.okOpen({effect:'togglePuff'});

  $("#example-1").on('mouseenter mouseleave',function(e){
    e.type == 'mouseenter' ? tooltip.open($(this).data('content'),{where:[this,0,-35]}) : tooltip.close();
  });
});

Modal Windows

Click Me 
          
var modal = $.okOpen({ effect:'fadeIn', modal: true}).addClass('modal');

$("#example-2").on('click',function(e){
  e.preventDefault();
  modal.open($(this).data('content'),{ where: 'center' });
});

$('.'+modal.options.overlayClass).click(function(e){ modal.close(); });
